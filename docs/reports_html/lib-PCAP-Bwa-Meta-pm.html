239         =head1 NAME
240        
241         PCAP::Bwa::Meta - Object to contain information about input files for mapping.
242        
243         =head2 Synopsis
244        
245         Please use the following to generate these objects:
246        
247         PCAP::Bwa::Meta::files_to_meta($tmp, \@files)
248        
249         This accepts bam, paired fastq and interleaved fastq. Fastq files can be gzip compressed.
250         The calling program should decide which type of files it is able to process.
251        
252         =head2 Constructor
253        
254         =over 4
255        
256         =item new
257        
258         Generates the object when called with hashref of options, please use
259         L<files_to_meta|PCAP::Bwa::Meta/files_to_meta> to generate the objects.
260        
261         =item files_to_meta
262        
263         Generates a set of PCAP::Bwa::Meta objects based on the input files.
264        
265         my $meta_set = PCAP::Bwa::Meta::files_to_meta($tmpdir, \@files);
266        
267         Returns a array reference of checked and pre-processed information.
268        
269         =head2 Object Methods
270        
271         =over 4
272        
273         =item in
274        
275         Can store BAM, FASTQ or INTERLEAVED FASTQ
276        
277         Retrieve the original file path. In the case of a paired FASTQ file this will be truncated back to
278         the common part of the filename so that both elements are represented by one object.
279        
280         FASTQ 1: /somepath/xxx_1.fq.gz
281         FASTQ 2: /somepath/xxx_2.fq.gz
282        
283         in = /somepath/xxx
284        
285         =item tstub
286        
287         A path for files generated from the input file represented here. The path is constructed from
288         'temp' and the rg_id.
289        
290         =item fastq
291        
292         Extension type of fastq file, see L<parser_fastq_filename|PCAP::Bwa::Meta/parse_fastq_filename>.
293        
294         =item paired_fq
295        
296         Flag to indicate that content of L<in()|PCAP::Bwa::Meta/in> is a stub for paired fastq rather
297         than an interleaved fastq, see L<parser_fastq_filename|PCAP::Bwa::Meta/parse_fastq_filename>.
298        
299         Added for future compatibility with files that can be used with 'BWA mem'.
300        
301         =item fastq
302        
303         When input is some form of fastq this returns the extension, otherwise undef.
304        
305         =item paired_fq
306        
307         When input is a paired fastq file this returns true.
308        
309         =item rg
310        
311         Returns the RG_ID that has been applied to this input.
312        
313         =item rg_header
314        
315         Call this with a hash of RG header tags to generate and retrieve the header, e.g.
316        
317         $meta->rg_header(q{\t}, -SN => 'some_sample_name', -LB => 'library_name');
318        
319         After generation you can retrieve with the required separator ('\t' for passing string to bwa on
320         command line, "\t" for direct use):
321        
322         print $meta->rg_header(qq{\t})."\n";
323        
324         $to_pass_to_command = $meta->rg_header(q{\t});
325        
326         SN is the only required element at present. See the Samtools L<specification|http://samtools.sourceforge.net/SAMv1.pdf>
327         for details.
328        
329         =back

373        
374         Expects file to end '_1' or '_2' for paired, assumed interleaved otherwise.
